cmake_minimum_required(VERSION 3.21)
project(SDL_Astro)

set(CMAKE_CXX_STANDARD 14)


include_directories(${PROJECT_SOURCE_DIR}/include)
if (WIN32)
    link_directories(${PROJECT_SOURCE_DIR}/lib/x64)
endif ()
add_executable(
        SDL_Astro
        src/main.cpp
        src/game.cpp
        src/textureManager.cpp
        src/player.cpp
        src/enemy.cpp
        src/projectiles.cpp
        src/Check_collision.cpp
        src/scoreManager.cpp include/Astro/scoreManager.h)

target_link_libraries(
  ${PROJECT_NAME}
  SDL2main
  SDL2
        SDL2_image
        SDL2_ttf.dll
)

add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_directory ${PROJECT_SOURCE_DIR}/lib/x64 ${PROJECT_SOURCE_DIR}/cmake-build-debug
)

add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_directory ${PROJECT_SOURCE_DIR}/assets ${PROJECT_SOURCE_DIR}/cmake-build-debug/assets
)
